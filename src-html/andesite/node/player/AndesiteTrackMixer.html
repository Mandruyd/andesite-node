001package andesite.node.player;
002
003import andesite.node.send.AudioProvider;
004
005import javax.annotation.CheckReturnValue;
006import javax.annotation.Nonnull;
007import java.util.Map;
008
009public interface AndesiteTrackMixer extends AudioProvider {
010    /**
011     * Returns a map of player id to mixer player.
012     *
013     * @return The players of the mixer.
014     */
015    @Nonnull
016    @CheckReturnValue
017    Map<String, MixerPlayer> players();
018    
019    /**
020     * Gets or creates a new player with the provided key. This key will be
021     * used to store this player in the map.
022     *
023     * @param key Key for the player.
024     *
025     * @return The current player for that key, or a new instance.
026     */
027    @Nonnull
028    @CheckReturnValue
029    MixerPlayer getPlayer(@Nonnull String key);
030    
031    /**
032     * Returns the current player for the provided key.
033     *
034     * @param key Key of the player.
035     *
036     * @return The current player for the provided key. May be null.
037     */
038    @CheckReturnValue
039    MixerPlayer getExistingPlayer(@Nonnull String key);
040    
041    /**
042     * Removes a player with the provided id.
043     *
044     * @param key Id of the player.
045     */
046    void removePlayer(@Nonnull String key);
047}
