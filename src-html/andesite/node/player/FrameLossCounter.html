001package andesite.node.player;
002
003import andesite.node.util.ByteRingBuffer;
004
005import javax.annotation.CheckReturnValue;
006import javax.annotation.Nonnull;
007
008public interface FrameLossCounter {
009    /**
010     * Packet count expected to be sent over a minute.
011     *
012     * <br><br>3000 packets with 20ms of audio each.
013     */
014    int EXPECTED_PACKET_COUNT_PER_MIN = (60 * 1000) / 20;
015    
016    /**
017     * Returns the amount of frames lost per second over the past minute.
018     * Each entry corresponds to one second. Ideally, all values should be 0.
019     *
020     * <br><br>This buffer should not be modified.
021     *
022     * <br><br>This data is only valid if {@link #isDataUsable()} returns true.
023     *
024     * @return The amount of frames lost over the past minute.
025     */
026    @Nonnull
027    @CheckReturnValue
028    ByteRingBuffer lastMinuteLoss();
029    
030    /**
031     * Returns the amount of frames sent per second over the past minute.
032     * Each entry corresponds to one second. Ideally, all values should be 30.
033     *
034     * <br><br>This buffer should not be modified.
035     *
036     * <br><br>This data is only valid if {@link #isDataUsable()} returns true.
037     *
038     * @return The amount of frames sent over the past minute.
039     */
040    @Nonnull
041    @CheckReturnValue
042    ByteRingBuffer lastMinuteSuccess();
043    
044    /**
045     * Returns whether or not enough data has been gathered for use.
046     *
047     * @return {@code true} if there's enough data stored.
048     */
049    @CheckReturnValue
050    boolean isDataUsable();
051}
