001package andesite.node.player.filter;
002
003import com.sedmelluq.discord.lavaplayer.filter.AudioFilter;
004import com.sedmelluq.discord.lavaplayer.filter.FilterChainBuilder;
005import com.sedmelluq.discord.lavaplayer.filter.PcmFilterFactory;
006import com.sedmelluq.discord.lavaplayer.filter.UniversalPcmAudioFilter;
007import com.sedmelluq.discord.lavaplayer.format.AudioDataFormat;
008import com.sedmelluq.discord.lavaplayer.track.AudioTrack;
009import io.vertx.core.json.JsonObject;
010
011import javax.annotation.CheckReturnValue;
012import javax.annotation.Nonnull;
013import javax.annotation.Nullable;
014import java.util.HashMap;
015import java.util.List;
016import java.util.Map;
017import java.util.Objects;
018import java.util.function.Supplier;
019
020public class FilterChainConfiguration {
021    private final EqualizerConfig equalizer = new EqualizerConfig();
022    private final KaraokeConfig karaoke = new KaraokeConfig();
023    private final TimescaleConfig timescale = new TimescaleConfig();
024    private final TremoloConfig tremolo = new TremoloConfig();
025    private final VibratoConfig vibrato = new VibratoConfig();
026    private final VolumeConfig volume = new VolumeConfig();
027    private final Map<Class<? extends Config>, Config> filters = new HashMap<>();
028    
029    public FilterChainConfiguration() {
030        filters.put(equalizer.getClass(), equalizer);
031        filters.put(karaoke.getClass(), karaoke);
032        filters.put(timescale.getClass(), timescale);
033        filters.put(tremolo.getClass(), tremolo);
034        filters.put(vibrato.getClass(), vibrato);
035        filters.put(volume.getClass(), volume);
036    }
037    
038    /**
039     * Returns true if a configuration of the provided class is present.
040     *
041     * @param clazz Class of the configuration.
042     *
043     * @return True if a configuration of the provided class is present.
044     */
045    public boolean hasConfig(Class<? extends Config> clazz) {
046        return filters.containsKey(clazz);
047    }
048    
049    /**
050     * Returns a configuration of the provided class, if it exists. May return null.
051     *
052     * @param clazz Class of the configuration.
053     * @param <T>   Type of the configuration.
054     *
055     * @return The existing instance, or null if there is none.
056     */
057    @SuppressWarnings("unchecked")
058    @CheckReturnValue
059    public <T extends Config> T config(@Nonnull Class<T> clazz) {
060        return (T) filters.get(clazz);
061    }
062    
063    /**
064     * Returns a configuration of the provided class if it exists, or creates
065     * a new instance of it with the provided supplier.
066     *
067     * @param clazz    Class of the configuration.
068     * @param supplier Supplier for creating a new instance of the configuration.
069     * @param <T>      Type of the configuration.
070     *
071     * @return An instance of the provided class stored. If none is stored, a new
072     * one is created and stored.
073     */
074    @SuppressWarnings("unchecked")
075    @CheckReturnValue
076    public <T extends Config> T config(@Nonnull Class<T> clazz, @Nonnull Supplier<T> supplier) {
077        return (T) filters.computeIfAbsent(clazz, __ -> {
078            var config = Objects.requireNonNull(supplier.get(), "Provided configuration may not be null");
079            if(!clazz.isInstance(config)) {
080                throw new IllegalArgumentException("Config not instance of provided class");
081            }
082            for(var c : filters.values()) {
083                if(c.name().equals(config.name())) {
084                    throw new IllegalArgumentException("Duplicate configuration name " + c.name());
085                }
086            }
087            return config;
088        });
089    }
090    
091    /**
092     * Returns whether or not this configuration has filters enabled.
093     *
094     * <br>This method returns true if any of the configurations reports
095     * it's enabled.
096     *
097     * @return True if this configuration is enabled.
098     */
099    @CheckReturnValue
100    public boolean isEnabled() {
101        for(var config : filters.values()) {
102            if(config.enabled()) return true;
103        }
104        return false;
105    }
106    
107    /**
108     * Returns a filter factory with the currently enabled filters.
109     *
110     * <br>If no configuration is enabled, this method returns null.
111     *
112     * @return A filter factory for the currently enabled filters,
113     * or null if none are enabled.
114     */
115    @Nullable
116    @CheckReturnValue
117    public PcmFilterFactory factory() {
118        return isEnabled() ? new Factory(this) : null;
119    }
120    
121    /**
122     * Encodes the state of this configuration and all filters in it.
123     *
124     * @return The encoded state.
125     */
126    @Nonnull
127    @CheckReturnValue
128    public JsonObject encode() {
129        var obj = new JsonObject();
130        for(Config config : filters.values()) {
131            obj.put(config.name(), config.encode().put("enabled", config.enabled()));
132        }
133        return obj;
134    }
135    
136    @Nonnull
137    @CheckReturnValue
138    public EqualizerConfig equalizer() {
139        return equalizer;
140    }
141    
142    @Nonnull
143    @CheckReturnValue
144    public KaraokeConfig karaoke() {
145        return karaoke;
146    }
147    
148    @Nonnull
149    @CheckReturnValue
150    public TimescaleConfig timescale() {
151        return timescale;
152    }
153    
154    @Nonnull
155    @CheckReturnValue
156    public TremoloConfig tremolo() {
157        return tremolo;
158    }
159    
160    @Nonnull
161    @CheckReturnValue
162    public VibratoConfig vibrato() {
163        return vibrato;
164    }
165    
166    @Nonnull
167    @CheckReturnValue
168    public VolumeConfig volume() {
169        return volume;
170    }
171    
172    private static class Factory implements PcmFilterFactory {
173        private final FilterChainConfiguration configuration;
174        
175        private Factory(FilterChainConfiguration configuration) {
176            this.configuration = configuration;
177        }
178        
179        @Override
180        public List<AudioFilter> buildChain(AudioTrack track, AudioDataFormat format, UniversalPcmAudioFilter output) {
181            var builder = new FilterChainBuilder();
182            builder.addFirst(output);
183            for(var config : configuration.filters.values()) {
184                var filter = config.enabled() ?
185                    config.create(format, builder.makeFirstFloat(format.channelCount)) //may return null
186                    : null;
187                if(filter != null) {
188                    builder.addFirst(filter);
189                }
190            }
191            var list = builder.build(null, format.channelCount).filters;
192            //remove output
193            return list.subList(1, list.size());
194        }
195    }
196}
