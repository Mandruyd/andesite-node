001package andesite.node.player.filter;
002
003import com.sedmelluq.discord.lavaplayer.filter.AudioFilter;
004import com.sedmelluq.discord.lavaplayer.filter.FloatPcmAudioFilter;
005import com.sedmelluq.discord.lavaplayer.filter.equalizer.Equalizer;
006import com.sedmelluq.discord.lavaplayer.format.AudioDataFormat;
007import io.vertx.core.json.JsonArray;
008import io.vertx.core.json.JsonObject;
009
010import javax.annotation.Nonnull;
011import javax.annotation.Nullable;
012
013public class EqualizerConfig implements Config {
014    private final float[] equalizerBands = new float[Equalizer.BAND_COUNT];
015    
016    public float getBand(int band) {
017        return equalizerBands[band];
018    }
019    
020    public void setBand(int band, float gain) {
021        equalizerBands[band] = gain;
022    }
023    
024    @Nonnull
025    @Override
026    public String name() {
027        return "equalizer";
028    }
029    
030    @Override
031    public boolean enabled() {
032        for(var band : equalizerBands) {
033            if(Config.isSet(band, 0f)) {
034                return true;
035            }
036        }
037        return false;
038    }
039    
040    @Nullable
041    @Override
042    public AudioFilter create(AudioDataFormat format, FloatPcmAudioFilter output) {
043        return Equalizer.isCompatible(format) ? new Equalizer(format.channelCount, output, equalizerBands) : null;
044    }
045    
046    @Nonnull
047    @Override
048    public JsonObject encode() {
049        var array = new JsonArray();
050        for(var band : equalizerBands) {
051            array.add(band);
052        }
053        return new JsonObject().put("bands", array);
054    }
055}
