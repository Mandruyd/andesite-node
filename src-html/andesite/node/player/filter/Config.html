001package andesite.node.player.filter;
002
003import com.sedmelluq.discord.lavaplayer.filter.AudioFilter;
004import com.sedmelluq.discord.lavaplayer.filter.FloatPcmAudioFilter;
005import com.sedmelluq.discord.lavaplayer.format.AudioDataFormat;
006import io.vertx.core.json.JsonObject;
007
008import javax.annotation.CheckReturnValue;
009import javax.annotation.Nonnull;
010import javax.annotation.Nullable;
011
012public interface Config {
013    /**
014     * Minimum absolute difference for floating point values. Values whose difference to the default
015     * value are smaller than this are considered equal to the default.
016     */
017    float MINIMUM_FP_DIFF = 0.01f;
018    
019    /**
020     * Returns the name of this filter, to be used in {@link FilterChainConfiguration#encode()}.
021     *
022     * @return A non null string representing this filter.
023     */
024    @CheckReturnValue
025    @Nonnull
026    String name();
027    
028    /**
029     * Returns whether or not this filter should be enabled.
030     *
031     * <br>For filters that are enabled, but unavailable, this method
032     * should return false.
033     *
034     * @return Whether or not the filter is enabled.
035     */
036    @CheckReturnValue
037    boolean enabled();
038    
039    /**
040     * Creates a new audio filter with the current settings.
041     *
042     * @param format Format of the audio.
043     * @param output Filter to write data to.
044     *
045     * @return A new audio filter.
046     */
047    @Nullable
048    @CheckReturnValue
049    AudioFilter create(AudioDataFormat format, FloatPcmAudioFilter output);
050    
051    /**
052     * Encodes the state of this configuration to send to clients.
053     *
054     * @return A json object containing the values of this configuration.
055     */
056    @Nonnull
057    @CheckReturnValue
058    JsonObject encode();
059    
060    /**
061     * Returns true if the difference between {@code value} and {@code defaultValue}
062     * is greater or equal to {@link #MINIMUM_FP_DIFF}.
063     *
064     * @param value        Value to check.
065     * @param defaultValue Default value.
066     *
067     * @return True if the difference is greater or equal to the minimum.
068     */
069    static boolean isSet(float value, float defaultValue) {
070        return Math.abs(value - defaultValue) >= MINIMUM_FP_DIFF;
071    }
072}
