001package andesite.node.player.filter;
002
003import com.github.natanbc.lavadsp.tremolo.TremoloPcmAudioFilter;
004import com.sedmelluq.discord.lavaplayer.filter.AudioFilter;
005import com.sedmelluq.discord.lavaplayer.filter.FloatPcmAudioFilter;
006import com.sedmelluq.discord.lavaplayer.format.AudioDataFormat;
007import io.vertx.core.json.JsonObject;
008
009import javax.annotation.Nonnull;
010import javax.annotation.Nullable;
011
012public class TremoloConfig implements Config {
013    private float frequency = 2f;
014    private float depth = 0.5f;
015    
016    public float frequency() {
017        return frequency;
018    }
019    
020    public void setFrequency(float frequency) {
021        if(frequency <= 0) {
022            throw new IllegalArgumentException("Frequency <= 0");
023        }
024        this.frequency = frequency;
025    }
026    
027    public float depth() {
028        return depth;
029    }
030    
031    public void setDepth(float depth) {
032        if(depth <= 0) {
033            throw new IllegalArgumentException("Depth <= 0");
034        }
035        if(depth > 1) {
036            throw new IllegalArgumentException("Depth > 1");
037        }
038        this.depth = depth;
039    }
040    
041    @Nonnull
042    @Override
043    public String name() {
044        return "tremolo";
045    }
046    
047    @Override
048    public boolean enabled() {
049        return Config.isSet(frequency, 2f) || Config.isSet(depth, 0.5f);
050    }
051    
052    @Nullable
053    @Override
054    public AudioFilter create(AudioDataFormat format, FloatPcmAudioFilter output) {
055        return new TremoloPcmAudioFilter(output, format.channelCount, format.sampleRate)
056            .setFrequency(frequency)
057            .setDepth(depth);
058    }
059    
060    @Nonnull
061    @Override
062    public JsonObject encode() {
063        return new JsonObject()
064            .put("frequency", frequency)
065            .put("depth", depth);
066    }
067}
