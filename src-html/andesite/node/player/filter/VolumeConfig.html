001package andesite.node.player.filter;
002
003import com.github.natanbc.lavadsp.volume.VolumePcmAudioFilter;
004import com.sedmelluq.discord.lavaplayer.filter.AudioFilter;
005import com.sedmelluq.discord.lavaplayer.filter.FloatPcmAudioFilter;
006import com.sedmelluq.discord.lavaplayer.format.AudioDataFormat;
007import io.vertx.core.json.JsonObject;
008
009import javax.annotation.Nonnull;
010import javax.annotation.Nullable;
011
012public class VolumeConfig implements Config {
013    private float volume = 1f;
014    
015    public float volume() {
016        return volume;
017    }
018    
019    public void setVolume(float volume) {
020        if(volume <= 0) {
021            throw new IllegalArgumentException("Volume <= 0.0");
022        }
023        if(volume > 5) {
024            throw new IllegalArgumentException("Volume > 5.0");
025        }
026        this.volume = volume;
027    }
028    
029    @Nonnull
030    @Override
031    public String name() {
032        return "volume";
033    }
034    
035    @Override
036    public boolean enabled() {
037        return Config.isSet(volume, 1f);
038    }
039    
040    @Nullable
041    @Override
042    public AudioFilter create(AudioDataFormat format, FloatPcmAudioFilter output) {
043        return new VolumePcmAudioFilter(output, format.channelCount)
044            .setVolume(volume);
045    }
046    
047    @Nonnull
048    @Override
049    public JsonObject encode() {
050        return new JsonObject()
051            .put("volume", volume);
052    }
053}
