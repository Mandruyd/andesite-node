001package andesite.node;
002
003import andesite.node.handler.WebSocketState;
004import com.sedmelluq.discord.lavaplayer.player.AudioPlayerManager;
005import gg.amy.singyeong.data.Dispatch;
006import io.vertx.core.json.JsonObject;
007import io.vertx.ext.web.Router;
008import io.vertx.ext.web.RoutingContext;
009
010import javax.annotation.CheckReturnValue;
011import javax.annotation.Nonnull;
012
013/**
014 * Entry point for a plugin. Defines the callbacks used to modify the node state
015 * or intercept commands.
016 */
017public interface Plugin {
018    /**
019     * Called when plugin loading is complete. This method is the first callback to be called
020     * and is guaranteed to run only once.
021     *
022     * <br><br>Blocking in this method is not a problem, but will delay node initialization.
023     *
024     * @param state State of the node.
025     *
026     * @see NodeState
027     */
028    default void init(@Nonnull NodeState state) {}
029    
030    /**
031     * Called to allow configuration of an audio player manager. May be called more than once with
032     * different player managers.
033     *
034     * <br><b>Plugins should not change the output format of the manager.</b>
035     *
036     * <br><br>Blocking in this method is not a problem, but will delay node initialization.
037     *
038     * @param state   State of the node.
039     * @param manager Player manager to configure.
040     *
041     * @see NodeState
042     * @see AudioPlayerManager
043     */
044    default void configurePlayerManager(@Nonnull NodeState state, @Nonnull AudioPlayerManager manager) {}
045    
046    /**
047     * Whether or not this plugin requires a router to work. If this returns true,
048     * a router will always be created, regardless of other settings. This
049     * method might not be called.
050     *
051     * <br>Even if this returns false, a router may still be created and
052     * {@link #configureRouter(NodeState, Router) configureRouter()} may
053     * be called. Returning true only ensures the creation, returning false
054     * doesn't stop it.
055     *
056     * <br>A router will be created if any of these conditions are met
057     * <ul>
058     * <li>HTTP or WebSocket transports are enabled</li>
059     * <li>Prometheus metrics are enabled</li>
060     * <li><b>Any</b> plugins returns true on this method</li>
061     * </ul>
062     *
063     * <br><br>Blocking in this method is not a problem, but will delay node initialization.
064     *
065     * @return Whether or not this plugin requires a router.
066     */
067    default boolean requiresRouter() {
068        return false;
069    }
070    
071    /**
072     * Called to allow injecting custom routes on the provided router.
073     * Only called if the HTTP server is enabled (see {@link #requiresRouter() requiresRouter()}
074     * for HTTP server creation conditions).
075     *
076     * <br><br>Blocking in this method is not a problem, but will delay node initialization.
077     *
078     * @param state  State of the node.
079     * @param router Root router used by the node.
080     *
081     * @see NodeState
082     * @see Router
083     */
084    default void configureRouter(@Nonnull NodeState state, @Nonnull Router router) {}
085    
086    /**
087     * Starts custom listeners for this plugin. If the default listeners are disabled
088     * and no plugin has custom listeners, the process will exit.
089     *
090     * <br><br>Blocking in this method is not a problem, but will delay node initialization.
091     *
092     * @param state State of the node.
093     *
094     * @return True if a listener was started.
095     *
096     * @see NodeState
097     */
098    default boolean startListeners(@Nonnull NodeState state) {
099        return false;
100    }
101    
102    /**
103     * Called when a REST request is received. Runs on the event loop, so <b>blocking should be avoided.</b>
104     *
105     * @param state   State of the node.
106     * @param context Context for the request.
107     *
108     * @return Whether or not other handlers should be called.
109     *
110     * @see NodeState
111     * @see RoutingContext
112     * @see HookResult
113     */
114    @Nonnull
115    @CheckReturnValue
116    default HookResult onRawHttpRequest(@Nonnull NodeState state, @Nonnull RoutingContext context) {
117        return HookResult.CALL_NEXT;
118    }
119    
120    /**
121     * Called when a websocket payload is received. Runs on the event loop, so <b>blocking should be avoided.</b>
122     *
123     * @param nodeState State of the node.
124     * @param wsState   State of the websocket connection.
125     * @param payload   Payload received.
126     *
127     * @return Whether or not other handlers should be called.
128     *
129     * @see NodeState
130     * @see WebSocketState
131     * @see HookResult
132     */
133    @Nonnull
134    @CheckReturnValue
135    default HookResult onRawWebSocketPayload(@Nonnull NodeState nodeState, @Nonnull WebSocketState wsState, @Nonnull JsonObject payload) {
136        return HookResult.CALL_NEXT;
137    }
138    
139    /**
140     * Called when a singyeong payload is received. Runs on the event loop, so <b>blocking should be avoided.</b>
141     *
142     * @param state   State of the node.
143     * @param payload Payload received plus metadata.
144     *
145     * @return Whether or not other handlers should be called.
146     *
147     * @see NodeState
148     * @see Dispatch
149     * @see HookResult
150     */
151    @Nonnull
152    @CheckReturnValue
153    default HookResult onRawSingyeongPayload(@Nonnull NodeState state, @Nonnull Dispatch payload) {
154        return HookResult.CALL_NEXT;
155    }
156    
157    /**
158     * Used to signal whether or not a payload should be dropped.
159     *
160     * Dropping a payload allows overriding other handlers or blocking a client.
161     */
162    enum HookResult {
163        /**
164         * Call next plugin, or the default andesite code if no more
165         * plugins remain.
166         */
167        CALL_NEXT,
168        /**
169         * Ignore this event.
170         */
171        ABORT
172    }
173}
