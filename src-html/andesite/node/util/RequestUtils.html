001package andesite.node.util;

003import com.sedmelluq.discord.lavaplayer.player.AudioPlayerManager;
004import com.sedmelluq.discord.lavaplayer.tools.io.MessageInput;
005import com.sedmelluq.discord.lavaplayer.tools.io.MessageOutput;
006import com.sedmelluq.discord.lavaplayer.track.AudioTrack;
007import io.vertx.core.json.JsonArray;
008import io.vertx.core.json.JsonObject;
009import io.vertx.ext.web.RoutingContext;

011import javax.annotation.CheckReturnValue;
012import javax.annotation.Nonnull;
013import javax.annotation.Nullable;
014import java.io.ByteArrayInputStream;
015import java.io.ByteArrayOutputStream;
016import java.io.IOException;
017import java.util.Base64;
018import java.util.function.Function;
019
020public class RequestUtils {
021    /**
022     * Attempts to find a password in a request.
023     *
024     * The password will be read from the following locations, in order, with
025     * the first one present being returned.
026     *
027     * <ul>
028     * <li>The {@code Authorization} header</li>
029     * <li>A websocket protocol which starts with {@code andesite-password:}
030     * (the returned password will have this prefix stripped)</li>
031     * <li>The {@code password} query param</li>
032     * </ul>
033     *
034     * If the websocket protocol is matched, it's value will be inserted into the
035     * {@code Sec-WebSocket-Protocol} header of the response.
036     *
037     * @param context Context where the password should be located.
038     *
039     * @return The password located, or null if nothing was found.
040     */
041    @Nullable
042    @CheckReturnValue
043    public static String findPassword(@Nonnull RoutingContext context) {
044        var authHeader = context.request().getHeader("Authorization");
045        if(authHeader != null) {
046            return authHeader;
047        }
048        
049        //allow browser access
050        //the browser websocket api doesn't support custom headers,
051        //so this hack is needed.
052        //browsers can use new WebSocket(url, "andesite-password:" + password)
053        var wsHeader = context.request().getHeader("Sec-WebSocket-Protocol");
054        if(wsHeader != null) {
055            var parts = wsHeader.split(",");
056            for(var part : parts) {
057                if(part.startsWith("andesite-password:")) {
058                    context.response().putHeader("Sec-WebSocket-Protocol", part);
059                    return part.substring("andesite-password:".length());
060                }
061            }
062        }
063        
064        var query = context.queryParam("password");
065        
066        return query.isEmpty() ? null : query.get(0);
067    }
068    
069    /**
070     * Decodes an audio track from it's base64 representation.
071     *
072     * @param playerManager Player manager used for decoding (must have the source manager enabled).
073     * @param base64        Base64 encoded track.
074     *
075     * @return The decoded track.
076     */
077    @Nullable
078    @CheckReturnValue
079    public static AudioTrack decodeTrack(@Nonnull AudioPlayerManager playerManager, @Nonnull String base64) {
080        try {
081            var v = playerManager.decodeTrack(new MessageInput(new ByteArrayInputStream(
082                Base64.getDecoder().decode(base64)
083            )));
084            return v == null ? null : v.decodedTrack;
085        } catch(IOException e) {
086            throw new AssertionError(e);
087        }
088    }
089    
090    /**
091     * Encodes the provided track to base64.
092     *
093     * @param playerManager Player manager used for encoding (must have the source manager enabled).
094     * @param track         Track to encode.
095     *
096     * @return Base64 encoded track.
097     */
098    @Nonnull
099    @CheckReturnValue
100    public static String trackString(@Nonnull AudioPlayerManager playerManager, @Nonnull AudioTrack track) {
101        var baos = new ByteArrayOutputStream();
102        try {
103            playerManager.encodeTrack(new MessageOutput(baos), track);
104        } catch(IOException e) {
105            throw new AssertionError(e);
106        }
107        return Base64.getEncoder().encodeToString(baos.toByteArray());
108    }
109    
110    /**
111     * Encodes a track into a json object, useful for sending to clients.
112     *
113     * @param playerManager Player manager used to encode the track.
114     * @param track         Track to encode.
115     *
116     * @return A json object containing information about the track.
117     */
118    @Nonnull
119    @CheckReturnValue
120    public static JsonObject encodeTrack(@Nonnull AudioPlayerManager playerManager, @Nonnull AudioTrack track) {
121        var info = track.getInfo();
122        return new JsonObject()
123            .put("track", trackString(playerManager, track))
124            .put("info", new JsonObject()
125                .put("class", track.getClass().getName())
126                .put("title", info.title)
127                .put("author", info.author)
128                .put("length", info.length)
129                .put("identifier", info.identifier)
130                .put("uri", info.uri)
131                .put("isStream", info.isStream)
132                .put("isSeekable", track.isSeekable())
133                .put("position", track.getPosition())
134            );
135    }
136    
137    /**
138     * Encodes the {@link RoutingContext#failure() failure} of the context.
139     * Equivalent to {@code encodeThrowable(context, context.failure())}
140     *
141     * @param context Context of the request.
142     *
143     * @return An encoded version of the exception.
144     *
145     * @see #encodeThrowable(RoutingContext, Throwable)
146     */
147    @Nonnull
148    @CheckReturnValue
149    public static JsonObject encodeFailure(@Nonnull RoutingContext context) {
150        return encodeThrowable(context, context.failure());
151    }
152    
153    /**
154     * Encodes the provided throwable, based on the context for deciding whether
155     * or not to use a shorter version with less details (omit stacktraces).
156     *
157     * @param context   Context for the request.
158     * @param throwable Error to encode.
159     *
160     * @return An encoded version of the exception.
161     */
162    @Nonnull
163    @CheckReturnValue
164    public static JsonObject encodeThrowable(@Nonnull RoutingContext context, @Nonnull Throwable throwable) {
165        return useShortMessage(context) ?
166            RequestUtils.encodeThrowableShort(throwable) :
167            RequestUtils.encodeThrowableDetailed(throwable);
168    }
169    
170    private static boolean useShortMessage(@Nonnull RoutingContext context) {
171        return context.request().getHeader("Andesite-Short-Errors") != null ||
172            context.queryParams().contains("shortErrors");
173    }
174    
175    /**
176     * Encodes a throwable with minimal details (omits stacktraces).
177     *
178     * @param throwable Error to encode.
179     *
180     * @return An encoded version of the exception.
181     */
182    @Nonnull
183    @CheckReturnValue
184    public static JsonObject encodeThrowableShort(@Nonnull Throwable throwable) {
185        var json = new JsonObject()
186            .put("class", throwable.getClass().getName())
187            .put("message", throwable.getMessage())
188            .put("suppressed", encodeArray(throwable.getSuppressed(), RequestUtils::encodeThrowableShort));
189        var cause = throwable.getCause();
190        if(cause != null) {
191            json.put("cause", encodeThrowableShort(cause));
192        } else {
193            json.putNull("cause");
194        }
195        return json;
196    }
197    
198    /**
199     * Encodes a throwable with as many details as possible. The resulting object may
200     * be very large.
201     *
202     * @param throwable Error to encode.
203     *
204     * @return An encoded version of the exception.
205     */
206    @Nonnull
207    @CheckReturnValue
208    public static JsonObject encodeThrowableDetailed(@Nonnull Throwable throwable) {
209        var json = new JsonObject()
210            .put("class", throwable.getClass().getName())
211            .put("message", throwable.getMessage())
212            .put("suppressed", encodeArray(throwable.getSuppressed(), RequestUtils::encodeThrowableDetailed))
213            .put("stack", encodeArray(throwable.getStackTrace(), RequestUtils::encodeStackFrame));
214        var cause = throwable.getCause();
215        if(cause != null) {
216            json.put("cause", encodeThrowableDetailed(cause));
217        } else {
218            json.putNull("cause");
219        }
220        return json;
221    }
222    
223    @Nonnull
224    @CheckReturnValue
225    private static JsonObject encodeStackFrame(@Nonnull StackTraceElement element) {
226        return new JsonObject()
227            .put("classLoader", element.getClassLoaderName())
228            .put("moduleName", element.getModuleName())
229            .put("moduleVersion", element.getModuleVersion())
230            .put("className", element.getClassName())
231            .put("methodName", element.getMethodName())
232            .put("fileName", element.getFileName())
233            .put("lineNumber", element.getLineNumber() < 0 ? null : element.getLineNumber())
234            .put("pretty", element.toString());
235    }
236    
237    @Nonnull
238    @CheckReturnValue
239    private static <T> JsonArray encodeArray(@Nonnull T[] array,
240                                             @Nonnull Function<T, JsonObject> serializeFunction) {
241        var json = new JsonArray();
242        for(var t : array) {
243            json.add(serializeFunction.apply(t));
244        }
245        return json;
246    }
