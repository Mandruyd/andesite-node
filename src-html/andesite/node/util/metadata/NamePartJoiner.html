001package andesite.node.util.metadata;
002
003import javax.annotation.CheckReturnValue;
004import javax.annotation.Nonnull;
005import java.util.List;
006import java.util.StringJoiner;
007import java.util.function.Function;
008import java.util.stream.Collectors;
009
010/**
011 * Joins a list of strings into a string, possibly transforming
012 * them or adding separators.
013 */
014public interface NamePartJoiner {
015    /**
016     * Joins the strings with a dash ({@code -}) character.
017     *
018     * <br>{@code [some, strings, go, here]} would become {@code some-strings-go-here}.
019     */
020    NamePartJoiner DASHED = joinWith("-");
021    
022    /**
023     * Joins the strings with a dash ({@code -}) character, after converting them to
024     * Title Case.
025     *
026     * <br>{@code [some, strings, go, here]} would become {@code Some-Strings-Go-Here}.
027     */
028    NamePartJoiner HTTP_HEADER = DASHED.afterMapping(NamePartJoiner::toTitleCase);
029    
030    /**
031     * Joins the strings after converting all but the first into Title Case.
032     *
033     * <br>{@code [some, strings, go, here]} would become {@code someStringsGoHere}.
034     */
035    NamePartJoiner LOWER_CAMEL_CASE = parts -> {
036        StringJoiner joiner = new StringJoiner("");
037        for(var it = parts.listIterator(); it.hasNext(); ) {
038            if(!it.hasPrevious()) {
039                joiner.add(it.next());
040            } else {
041                joiner.add(toTitleCase(it.next()));
042            }
043        }
044        return joiner.toString();
045    };
046    
047    /**
048     * Joins the provided parts into a string. The joiner is
049     * free to apply any transformation it wants to the strings,
050     * and join them however it wants.
051     *
052     * @param parts Parts to join. All strings in this list should
053     *              be in lower case.
054     *
055     * @return The resulting string.
056     */
057    @Nonnull
058    @CheckReturnValue
059    String join(@Nonnull List<String> parts);
060    
061    /**
062     * Returns a new joiner that will join the strings resulting
063     * from applying the provided function to each part.
064     *
065     * @param mapper Maps the parts before joining.
066     *
067     * @return A joiner that maps the parts before calling the join method.
068     */
069    @Nonnull
070    @CheckReturnValue
071    default NamePartJoiner afterMapping(@Nonnull Function<String, String> mapper) {
072        return parts -> join(parts.stream().map(mapper).collect(Collectors.toList()));
073    }
074    
075    /**
076     * Joins the parts with the provided separator.
077     *
078     * @param separator Separator for the parts.
079     *
080     * @return Joiner that separates the parts with the given separator.
081     */
082    static NamePartJoiner joinWith(@Nonnull String separator) {
083        return parts -> {
084            StringJoiner joiner = new StringJoiner(separator);
085            parts.forEach(joiner::add);
086            return joiner.toString();
087        };
088    }
089    
090    private static String toTitleCase(String s) {
091        return Character.toUpperCase(s.charAt(0)) + s.substring(1);
092    }
093}
