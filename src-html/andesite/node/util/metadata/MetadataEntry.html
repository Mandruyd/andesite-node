001package andesite.node.util.metadata;
002
003import javax.annotation.CheckReturnValue;
004import javax.annotation.Nonnull;
005import java.util.ArrayList;
006import java.util.Collection;
007import java.util.List;
008
009public class MetadataEntry {
010    private final MetadataType type;
011    private final Object value;
012    
013    private MetadataEntry(@Nonnull MetadataType type, @Nonnull Object value) {
014        this.type = type;
015        this.value = value;
016    }
017    
018    @Nonnull
019    @CheckReturnValue
020    public MetadataType type() {
021        return type;
022    }
023    
024    @Nonnull
025    @CheckReturnValue
026    public Object rawValue() {
027        return value;
028    }
029    
030    @Nonnull
031    @CheckReturnValue
032    public String asString() {
033        checkType(MetadataType.STRING, MetadataType.VERSION);
034        return (String) value;
035    }
036    
037    @CheckReturnValue
038    public int asInteger() {
039        checkType(MetadataType.INTEGER);
040        return (int) value;
041    }
042    
043    @Nonnull
044    @CheckReturnValue
045    @SuppressWarnings("unchecked")
046    public List<String> asStringList() {
047        checkType(MetadataType.STRING_LIST);
048        return (List<String>) value;
049    }
050    
051    @Nonnull
052    @CheckReturnValue
053    public String asVersion() {
054        checkType(MetadataType.VERSION);
055        return (String) value;
056    }
057    
058    @Nonnull
059    @CheckReturnValue
060    public static MetadataEntry string(@Nonnull String value) {
061        return new MetadataEntry(MetadataType.STRING, value);
062    }
063    
064    @Nonnull
065    @CheckReturnValue
066    public static MetadataEntry integer(int value) {
067        return new MetadataEntry(MetadataType.INTEGER, value);
068    }
069    
070    @Nonnull
071    @CheckReturnValue
072    public static MetadataEntry stringList(@Nonnull Collection<String> value) {
073        return new MetadataEntry(MetadataType.STRING_LIST, value instanceof List ? value : new ArrayList<>(value));
074    }
075    
076    @Nonnull
077    @CheckReturnValue
078    public static MetadataEntry version(@Nonnull String value) {
079        return new MetadataEntry(MetadataType.VERSION, value);
080    }
081    
082    @Override
083    public int hashCode() {
084        return value.hashCode();
085    }
086    
087    @Override
088    public boolean equals(Object obj) {
089        return obj instanceof MetadataEntry && value.equals(((MetadataEntry) obj).value);
090    }
091    
092    @Override
093    public String toString() {
094        switch(type) {
095            case STRING:
096            case VERSION:
097                return (String) value;
098            case INTEGER:
099                return value.toString();
100            case STRING_LIST:
101                return String.join(",", asStringList());
102            default:
103                throw new AssertionError();
104        }
105    }
106    
107    private void checkType(@Nonnull MetadataType type) {
108        if(this.type != type) {
109            throw new IllegalStateException("Expected type to be " + type + ", was " + this.type);
110        }
111    }
112    
113    private void checkType(@Nonnull MetadataType t1, @Nonnull MetadataType t2) {
114        if(this.type != t1 && this.type != t2) {
115            throw new IllegalStateException("Expected type to be " + t1 + " or " + t2 + ", was " + this.type);
116        }
117    }
118}
