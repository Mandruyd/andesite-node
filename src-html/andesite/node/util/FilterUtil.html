001package andesite.node.util;
002
003
004import com.github.natanbc.lavadsp.natives.TimescaleNativeLibLoader;
005
006/**
007 * Utility class listing what filters are available. Unavailable filters may still be enabled,
008 * but won't be applied.
009 */
010public class FilterUtil {
011    public static final boolean TIMESCALE_AVAILABLE = tryLoad(TimescaleNativeLibLoader::loadTimescaleLibrary);
012    
013    private static boolean tryLoad(Runnable load) {
014        try {
015            load.run();
016            return true;
017        } catch(Throwable error) {
018            return false;
019        }
020    }
021}
