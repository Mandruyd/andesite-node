001package andesite.node;
002
003import andesite.node.event.EventDispatcher;
004import andesite.node.handler.AndesiteRequestHandler;
005import andesite.node.player.AndesitePlayer;
006import andesite.node.send.AudioHandler;
007import com.sedmelluq.discord.lavaplayer.player.AudioPlayerManager;
008import com.typesafe.config.Config;
009import io.vertx.core.Vertx;
010
011import javax.annotation.CheckReturnValue;
012import javax.annotation.Nonnull;
013import javax.annotation.Nullable;
014import java.lang.ref.Cleaner;
015import java.util.Map;
016import java.util.stream.Stream;
017
018/**
019 * Provides access to andesite objects for plugins.
020 */
021public interface NodeState {
022    /**
023     * Config instance used by the node. Plugins should read configurations from this object,
024     * which loads from the appropriate sources.
025     *
026     * @return The node configuration.
027     */
028    @Nonnull
029    @CheckReturnValue
030    Config config();
031    
032    /**
033     * Vertx instance used by the node.
034     *
035     * @return The vertx instance.
036     */
037    @Nonnull
038    @CheckReturnValue
039    Vertx vertx();
040    
041    /**
042     * The request handler used by the node.
043     *
044     * @return The request handler.
045     */
046    @Nonnull
047    @CheckReturnValue
048    AndesiteRequestHandler requestHandler();
049    
050    /**
051     * The regular player manager, which returns opus data.
052     *
053     * @return The regular player manager.
054     */
055    @Nonnull
056    @CheckReturnValue
057    AudioPlayerManager audioPlayerManager();
058    
059    /**
060     * The PCM player manager, which returns pcm data. Used by the track mixer for better performance.
061     *
062     * @return The PCM player manager.
063     */
064    @Nonnull
065    @CheckReturnValue
066    AudioPlayerManager pcmAudioPlayerManager();
067    
068    /**
069     * The event dispatcher used by the node. You can use this to dynamically register/unregister listeners.
070     *
071     * @return The event dispatcher.
072     */
073    @Nonnull
074    @CheckReturnValue
075    EventDispatcher dispatcher();
076    
077    /**
078     * The audio handler used by the node. Allows lower level configuration of the player.
079     *
080     * @return The audio handler.
081     */
082    @Nonnull
083    @CheckReturnValue
084    AudioHandler audioHandler();
085    
086    /**
087     * Returns a map of the players owned by the provided user. This map is thread safe and will
088     * be updated when players are created/destroyed.
089     *
090     * @param userId User id that owns the map.
091     *
092     * @return The map of guild id {@literal ->} player for the provided user.
093     *
094     * @see AndesitePlayer
095     */
096    @Nonnull
097    @CheckReturnValue
098    Map<String, ? extends AndesitePlayer> playerMap(@Nonnull String userId);
099    
100    /**
101     * Gets or creates a player for the provided user and guild. Equivalent to
102     * {@code playerMap(userId).computeIfAbsent(guildId, createPlayer);}.
103     *
104     * @param userId  User id for the player.
105     * @param guildId Guild id for the player.
106     *
107     * @return The current player, if it exists, or a new player.
108     *
109     * @see AndesitePlayer
110     */
111    @Nonnull
112    @CheckReturnValue
113    AndesitePlayer getPlayer(@Nonnull String userId, @Nonnull String guildId);
114    
115    /**
116     * Returns the existing player for the provided user and guild, or null if it doesn't exist.
117     * Equivalent to {@code playerMap(userId).get(guildId)}.
118     *
119     * @param userId  User id for the player.
120     * @param guildId Guild id for the player.
121     *
122     * @return The current player.
123     *
124     * @see AndesitePlayer
125     */
126    @Nullable
127    @CheckReturnValue
128    AndesitePlayer getExistingPlayer(@Nonnull String userId, @Nonnull String guildId);
129    
130    /**
131     * Removes and stops the player for the provided user and guild, returning it.
132     * Returns null if there was no player.
133     *
134     * @param userId  User id for the player.
135     * @param guildId Guild id for the player.
136     *
137     * @return The removed player.
138     *
139     * @see AndesitePlayer
140     */
141    @Nullable
142    AndesitePlayer removePlayer(@Nonnull String userId, @Nonnull String guildId);
143    
144    /**
145     * Stream containing all existing players, for all users and guilds.
146     *
147     * @return All existing players.
148     *
149     * @see AndesitePlayer
150     */
151    @Nonnull
152    @CheckReturnValue
153    Stream<? extends AndesitePlayer> allPlayers();
154    
155    /**
156     * Cleaner used to register actions on garbage collection of objects.
157     * Plugins should use this cleaner instead of creating their own.
158     *
159     * @return The cleaner instance for the node.
160     *
161     * @see Cleaner
162     */
163    @Nonnull
164    @CheckReturnValue
165    Cleaner cleaner();
166}
