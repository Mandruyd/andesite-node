001package andesite.node.send;
002
003import javax.annotation.Nonnull;
004import javax.annotation.Nullable;
005
006public interface AudioHandler {
007    /**
008     * Sets the provider for the given user and guild. A null provider will remove
009     * the existing provider, if it exists.
010     *
011     * <br>If there is already a provider set, it's {@link AudioProvider#close() close}
012     * method will be called.
013     *
014     * @param userId   User id for the provider.
015     * @param guildId  Guild id for the provider.
016     * @param provider Provider to set.
017     */
018    void setProvider(@Nonnull String userId, @Nonnull String guildId, @Nullable AudioProvider provider);
019    
020    /**
021     * Handles a voice update.
022     *
023     * @param userId    User id for the update.
024     * @param guildId   Guild id for the update.
025     * @param sessionId Session id of the voice state.
026     * @param endpoint  Endpoint of the voice server.
027     * @param token     Token of the voice connection.
028     */
029    void handleVoiceUpdate(@Nonnull String userId, @Nonnull String guildId,
030                           @Nonnull String sessionId, @Nonnull String endpoint,
031                           @Nonnull String token);
032    
033    /**
034     * Closes a connection for the given user and guild. If the connection
035     * has a provider set, it's {@link AudioProvider#close() close} method
036     * will be called.
037     *
038     * @param userId  User id of the connection.
039     * @param guildId Guild id of the connection.
040     */
041    void closeConnection(@Nonnull String userId, @Nonnull String guildId);
042}
