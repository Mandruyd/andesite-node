001package andesite.node.send;
002
003import javax.annotation.CheckReturnValue;
004import javax.annotation.Nonnull;
005import java.nio.ByteBuffer;
006
007/**
008 * Used to provide audio for an {@link AudioHandler}. The format the provided audio should be
009 * depends on the handler. Currently only opus audio is supported.
010 */
011public interface AudioProvider extends AutoCloseable {
012    /**
013     * Attempts to load the next audio frame, returning true if there is one.
014     *
015     * <br>If this method returns true, the frame data should be returned in the next
016     * {@link #provide() provide} call.
017     *
018     * @return Whether or not an audio frame was loaded.
019     */
020    @CheckReturnValue
021    boolean canProvide();
022    
023    /**
024     * Returns the data for the latest frame.
025     *
026     * @return Buffer containing the frame's data.
027     */
028    @Nonnull
029    @CheckReturnValue
030    ByteBuffer provide();
031    
032    /**
033     * Runs any cleanup actions. Called when this provider is removed with
034     * {@link AudioHandler#setProvider(String, String, AudioProvider)} or
035     * {@link AudioHandler#closeConnection(String, String)}.
036     *
037     * <br>This method may be called more than once if this provider is added
038     * again to the audio handler, but will be called only once each time it is removed.
039     */
040    void close();
041}
